282 float Gaussian::function(const float lambda_) const
283 {
284  float mu_ = static_cast<float>(this->mu);
285  float sigma_ = static_cast<float>(this->sigma);
286  float x = (lambda_ - mu_) / sigma_;
287  return (0.5 * M_SQRT1_2 * M_2_SQRTPI / sigma_) * exp(-0.5 * x * x);
288 }
