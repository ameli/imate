70 cdef double _matern_kernel(
71  const double x,
72  const double param) nogil:
73  """
74  Computes the Matern class correlation function for a given Euclidean
75  distance of two spatial points.
76 
77  The Matern correlation function defined by
78 
79  .. math::
80  K(\\boldsymbol{x},\\boldsymbol{x}'|\\rho,\\nu) =
81  \\frac{2^{1-\\nu}}{\\Gamma(\\nu)}
82  \\left( \\sqrt{2 \\nu} \\frac{\\| \\boldsymbol{x} -
83  \\boldsymbol{x}' \\|}{\\rho} \\right)
84  K_{\\nu}\\left(\\sqrt{2 \\nu} \\frac{\\|\\boldsymbol{x} -
85  \\boldsymbol{x}' \\|}{\\rho} \\right)
86 
87  where
88 
89  * :math:`\\rho` is the correlation scale of the function,
90  * :math:`\\Gamma` is the Gamma function,
91  * :math:`\\| \\cdot \\|` is the Euclidean distance,
92  * :math:`\\nu` is the smoothness parameter.
93  * :math:`K_{\\nu}` is the modified Bessel function of the second kind
94  of order :math:`\\nu`
95 
96  .. warning::
97 
98  When the distance :math:`\\| \\boldsymbol{x} - \\boldsymbol{x}' \\|` is
99  zero, the correlation function produces :math:`\\frac{0}{0}` but in the
100  limit, the correlation function is :math:`1`. If the distance is not
101  exactly zero, but close to zero, this function might produce unstable
102  results.
103 
104  In this function, it is assumed that :math:`\\nu = \\frac{5}{2}`, and the
105  Matern correlation in this case can be represented by:
106 
107  .. math::
108  K(\\boldsymbol{x},\\boldsymbol{x}'|\\rho,\\nu) =
109  \\left( 1 + \\sqrt{5} \\frac{\\| \\boldsymbol{x} -
110  \\boldsymbol{x}'\\|}{\\rho} + \\frac{5}{3} \\frac{\\| \\boldsymbol{x} -
111  \\boldsymbol{x}'\\|^2}{\\rho^2} \\right)
112  \\exp \\left( -\\sqrt{5} \\frac{\\| \\boldsymbol{x} -
113  \\boldsymbol{x}'\\|}{\\rho} \\right)
114 
115  :param x: The distance that represents the Euclidean distance between
116  mutual points.
117  :type x: ndarray
118 
119  :param nu: The parameter :math:`\\nu` of the Matern kernel.
120  :type nu: double
121 
122  :return: Matern correlation
123  :rtype: double
124  """
125 
126  # scaled distance
127  cdef double nu = param
128  cdef double correlation
129 
130  if x == 0:
131  correlation = 1.0
132  else:
133  if nu == 0.5:
134  correlation = exp(-x)
135  elif nu == 1.5:
136  correlation = (1.0 + sqrt(3.0) * x) * exp(-sqrt(3.0) * x)
137  elif nu == 2.5:
138  correlation = (1.0 + sqrt(5.0) * x + (5.0 / 3.0) * (x**2)) * \
139  exp(-sqrt(5.0) * x)
140  elif nu < 100:
141 
142  # Change zero elements of y to a dummy number, to avoid
143  # multiplication of zero by Inf in Bessel function below
144  correlation = ((2.0**(1.0-nu)) / gamma(nu)) * \
145  ((sqrt(2.0*nu) * x)**nu) * kv(nu, sqrt(2.0*nu)*x)
146 
147  else:
148  # For nu > 100, assume nu is Inf. In this case, Matern function
149  # approaches Gaussian kernel
150  correlation = exp(-0.5*x**2)
151 
152  if isnan(correlation):
153  printf('correlation is nan.\n')
154  if isinf(correlation):
155  printf('correlation is inf.\n')
156 
157  return correlation

204  const double x,
205  const double param) nogil:
206  """
207  Rational quadratic kernel.
208  """
209 
210  cdef double alpha = param
211  cdef double correlation = 1.0 + x**2 / (2.0 * alpha)

223 @boundscheck(False)
224 @wraparound(False)
225 cdef double euclidean_distance(
226  const double[:] point1,
227  const double[:] point2,
228  const double scale,
229  const int dimension) nogil:
230  """
231  Returns the Euclidean distance between two points.
232 
233  :param point1: 1D array of the coordinates of a point
234  :type point1: cython memoryview (double)
235 
236  :param point2: 1D array of the coordinates of a point
237  :type point2: cython memoryview (double)
238 
239  :param scale: A parameter to scale the distance
240  :type scale: double
241 
242  :param dimension: Dimension of the coordinates of the points.
243  :type dimension: int
244 
245  :return: Euclidean distance betwrrn point1 and point2
246  :rtype: double
247  """
248 
249  cdef double distance2 = 0
250  cdef int dim
251 
252  for dim in range(dimension):
253  distance2 += (point1[dim] - point2[dim])**2
254 
255  return sqrt(distance2) / scale
