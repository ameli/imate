84 {
85  assert(num_threads_ > 0);
86 
87  this->num_threads = num_threads_;
88  this->xoshiro_256_star_star = new Xoshiro256StarStar[this->num_threads];
89 
90  for (int i=0; i < this->num_threads; ++i)
91  {
92  // Repeate jump j times to have different initial state for each thread
93  // This is the main purpose of this class.
94  for (int j=0; j < i+1; ++j)
95  {
96  this->xoshiro_256_star_star[i].jump();
97  }
98  }
