22 def check_convergence(
23  samples,
24  min_num_samples,
25  processed_samples_indices,
26  num_processed_samples,
27  confidence_level,
28  error_atol,
29  error_rtol):
30  """
31  Checks if the standard deviation of the set of the cumulative averages of
32  trace estimators converged below the given tolerance.
33 
34  The convergence criterion for each trace inquiry is if:
35 
36  standard_deviation < max(rtol * average[i], atol)
37 
38  where `error_rtol` and `error_atol` are relative and absolute tolerances,
39  respectively. If this criterion is satisfied for *all* trace inquiries,
40  this function returns `1`, otherwise `0`.
41  """
42 
43  # If number of processed samples are not enough, set to not converged yet.
44  # This is essential since in the first few iterations, the standard
45  # deviation of the cumulative averages are still too small.
46  if num_processed_samples < min_num_samples:
47 
48  # Skip computing error. Fill outputs with trivial initial values
49  converged = 0
50  num_samples_used = num_processed_samples
51  return converged, num_samples_used
52 
53  # Quantile of normal distribution (usually known as the "z" coefficient)
54  standard_z_score = numpy.sqrt(2) * erfinv(confidence_level)
55 
56  # mean using all processed rows of j-th column
57  mean = numpy.mean(
58  samples[processed_samples_indices[:num_processed_samples]])
59 
60  # std using all processed rows of j-th column
61  if num_processed_samples > 1:
62  std = numpy.std(
63  samples[processed_samples_indices[:num_processed_samples]])
64  else:
65  std = numpy.inf
66 
67  # Compute error based of std and confidence level
68  error = standard_z_score * std / numpy.sqrt(num_processed_samples)
69 
70  # Check error with atol and rtol to find if converged
71  if error < numpy.max([error_atol, error_rtol*mean]):
72  converged = 1
73  else:
74  converged = 0
75 
76  # Update how many samples used so far to average
77  num_samples_used = num_processed_samples
78 
79  return converged, num_samples_used

86 def average_estimates(
87  confidence_level,
88  outlier_significance_level,
89  max_num_samples,
90  num_samples_used,
91  processed_samples_indices,
92  samples):
93  """
94  Averages the estimates of trace. Removes outliers and reevaluates the error
95  to take into account for the removal of the outliers.
96 
97  The elimination of outliers does not affect the elements of samples array,
98  rather it only affects the reevaluation of trac and error arrays.
99  """
100 
101  # Flag which samples are outliers
102  outlier_indices = numpy.zeros((max_num_samples, ), dtype=int)
103 
104  # Quantile of normal distribution (usually known as the "z" coefficient)
105  error_z_score = numpy.sqrt(2) * erfinv(confidence_level)
106 
107  # Confidence level of outlier is the complement of significance level
108  outlier_confidence_level = 1.0 - outlier_significance_level
109 
110  # Quantile of normal distribution area where is not considered as outlier
111  outlier_z_score = numpy.sqrt(2.0) * erfinv(outlier_confidence_level)
112 
113  # Compute mean of samples
114  mean = numpy.mean(
115  samples[processed_samples_indices[:num_samples_used]])
116 
117  # Compute std of samples
118  if num_samples_used > 1:
119  std = numpy.std(
120  samples[processed_samples_indices[:num_samples_used]])
121  else:
122  std = numpy.inf
123 
124  # Outlier half interval
125  outlier_half_interval = outlier_z_score * std
126 
127  # Difference of each element from
128  num_outliers = 0
129  for i in range(num_samples_used):
130 
131  mean_discrepancy = samples[processed_samples_indices[i]] - mean
132  if numpy.abs(mean_discrepancy) > outlier_half_interval:
133 
134  # Outlier detected
135  outlier_indices[i] = 1
136  num_outliers += 1
137 
138  # Reevaluate mean but leave out outliers
139  summand = 0.0
140  for i in range(num_samples_used):
141  if outlier_indices[i] == 0:
142  summand += samples[processed_samples_indices[i]]
143  mean = summand / (num_samples_used - num_outliers)
144 
145  # Reevaluate std but leave out outliers
146  if num_samples_used > 1 + num_outliers:
147  summand = 0.0
148  for i in range(num_samples_used):
149  if outlier_indices[i] == 0:
150  mean_discrepancy = samples[processed_samples_indices[i]] - mean
151  summand += mean_discrepancy * mean_discrepancy
152 
153  std = numpy.sqrt(summand/(num_samples_used - num_outliers - 1.0))
154  else:
155  std = numpy.inf
156 
157  # trace and its error
158  trace = mean
159  error = error_z_score * std / numpy.sqrt(num_samples_used - num_outliers)
160 
161  return trace, error, num_outliers
