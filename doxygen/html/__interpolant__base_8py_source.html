265  def upper_bound(self, t):
266  """
267  Upper bound of the function :math:`t \\mapsto \\mathrm{trace} \\left(
268  (\\mathbf{A} + t \\mathbf{B})^{-1} \\right)`.
269 
270  The lower bound is given by Remark 2.2 of [Ameli-2020]_ as
271 
272  .. math::
273 
274  \\mathrm{trace}((\\mathbf{A}+t\\mathbf{B})^{-1}) \\geq
275  \\frac{n^2}{\\mathrm{trace}(\\mathbf{A}) +
276  \\mathrm{trace}(t \\mathbf{B})}
277 
278  :param t: Inquiry points
279  :type t: float or numpy.ndarray
280 
281  :return: Lower bound of the trace of inverse at inquiry points.
282  :rtype: float or numpy.ndarray
283  """
284 
285  if self.pp != -1:
286  raise ValueError('Lower bound is only available for "p=-1".')
287 
288  # Trace of A and B
289  trace_A = numpy.trace(self.AA)
290  trace_B = numpy.trace(self.BB)
291 
292  # Lower bound of trace of A+tB
293  trace_ub = (self.nn**2)/(trace_A + t*trace_B)
294  schatten_ub = (trace_ub / self.nn)**(1.0/self.pp)
295 
296  return schatten_ub

322 
323  :param t: Inquiry points

326  :return: Upper bound of the trace of inverse at inquiry points.
327  :rtype: float or numpy.ndarray
