23 # generate random column vectors
24 # ==============================
25 
26 cdef void generate_random_column_vectors(
27  DataType* vectors,
28  const LongIndexType vector_size,
29  const IndexType num_vectors,
30  const IndexType orthogonalize,
31  const IndexType num_threads):
32  """
33  Generates a set of random column vectors using Rademacher distribution. The
34  column vectors are normalized to unit norm. If desired, the vectors can
35  also be orthogonalized by setting ``orthogonalize`` flag to a non-zero
36  value. The computation is parallelized using OpenMP over the number of
37  column vectors.
38 
39  :param vectors: 2D array of vectors of size ``(vector_size, num_vectors)``.
40  This array will be written in-place and serves as the output. Note this
41  is Fortran ordering, meaning that the first index is contiguous. To
42  refer to the j-th column vector, use ``&vectors[0][j]``.
43  :type vectors: pointer (DataType)
44 
45  :param vector_size: Number of rows of vectors array.
46  :type vector_size: LongIndexType
47 
48  :param num_vectors: Number of columns of vectors array.
49  :type num_vectors: IndexType
50 
51  :param orthogonalize: A flag, when set to a non-zero value, the vectors are
52  orthogonalized using modified Gram-Schmidt process. Otherwise, the
53  generated vectors remain enact.
54  :param orthogonalize: IndexType
55 
56  :param num_threads: Number of OpenMP parallel threads
57  :type num_threads: IndexType
58  """
59 
60  # Generate random array
61  cdef LongIndexType vectors_size = vector_size * num_vectors
62  py_generate_random_array(vectors, vectors_size, num_threads)
63 
64  # Orthogonalize (optional). This section is not parallel (must be serial)
65  if orthogonalize:
66  orthogonalize_vectors(vectors, vector_size, num_vectors)
67 
68  # Set the number of threads
69  openmp.omp_set_num_threads(num_threads)
70 
71  # Using max possible chunk size for parallel schedules
72  cdef IndexType chunk_size = int((<DataType> num_vectors) / num_threads)
73  if chunk_size < 1:
74  chunk_size = 1
75 
76  # Normalize (necessary)
77  cdef IndexType j
78  with nogil, parallel():
79  for j in prange(num_vectors, schedule='static', chunksize=chunk_size):
80  cVectorOperations[DataType].normalize_vector_in_place(
81  &vectors[j*vector_size], vector_size)
