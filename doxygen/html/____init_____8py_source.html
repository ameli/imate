14 def _check_import():
15  """
16  The following two issues (see if conditions below) only happen in a cython
17  package and if the package is build without --inplace option. This is
18  because without the --inplace option, the *.so files will be built inside
19  the /build directory (not in the same directory of the source code where
20  *.pyx files are). On the other hand, when the user's directory is in the
21  parent directory of the package, this path will be the first path on the
22  sys.path. Thus, it looks for the package in the source-code directory, not
23  where it was built or installed. But, because the built is outside of the
24  source (recall no --inplace), it cannot find the *.so files.
25 
26  To resolve this issue:
27  1. Either build the package with --inplace option.
28  2. Change the current directory, or the directory of the script that you
29  are running out of the source code.
30  """
31 
32  import sys
33  import os
34 
35  # Find the current directory of user (where the user calls an executable)
36  _user_current_dir = os.getcwd()
37 
38  # Find executable directory (this is where the *.py executable is)
39  _executable_file = os.path.abspath(sys.argv[0])
40  _executable_dir = os.path.dirname(_executable_file)
41 
42  # Find the project directory (second parent directory of this script)
43  _package_dir = os.path.dirname(os.path.realpath(__file__)) # It is: ../
44  _project_dir = os.path.dirname(_package_dir) # It is: ../../
45 
46  if (_user_current_dir == _project_dir):
47  raise RuntimeError('You are in the source-code directory of this ' +
48  'package. Importing the package will fail. To ' +
49  'resolve this issue, consider changing the ' +
50  'current directory outside of the directory of ' +
51  'the source-code of this package. Your current' +
52  'directory is: %s.' % _user_current_dir)
53 
54  elif (_executable_dir == _project_dir):
55  raise RuntimeError('You are running a script in the source-code ' +
56  'directory of this package. Importing the ' +
57  'package will fail. To resolve this issue, ' +
58  'consider changing the script directory outside' +
59  'of the directory of the source-code of this' +
60  'package. Your current directory is: %s.'
61  % _executable_dir)
