29 class ChebyshevRationalFunctionsMethod(InterpolantBase):
30  """
31  Computes the trace of inverse of an invertible matrix :math:`\\mathbf{A} +
32  t \\mathbf{B}` using an interpolation scheme based on rational polynomial
33  functions (see details below).
34 
35  **Class Inheritance:**
36 
37  .. inheritance-diagram::
38  imate.InterpolateTraceinv.RadialBasisFunctionsMethod
39  :parts: 1
40 
41  :param A: Invertible matrix, can be either dense or sparse matrix.
42  :type A: numpy.ndarray
43 
44  :param B: Invertible matrix, can be either dense or sparse matrix.
45  :type B: numpy.ndarray
46 
47  :param options: A dictionary of input arguments for
48  :mod:`imate.traceinv.traceinv` module.
49  :type options: dict
50 
51  :param verbose: If ``True``, prints some information on the computation
52  process. Default is ``False``.
53  :type verbose: bool
54 
55  :param function_type: Can be ``1``, ``2``, or ``3``, which defines
56  different radial basis functions (see details below).
57  :type function_type: int
58 
59  **Interpolation Method**
60 
61  Define the function
62 
63  .. math::
64 
65  \\tau(t) = \\frac{\\mathrm{trace}\\left( (\\mathbf{A} +
66  t \\mathbf{B})^{-1} \\right)}{\\mathrm{trace}(\\mathbf{B}^{-1})}
67 
68  and :math:`\\tau_0 = \\tau(0)`. Then, we approximate :math:`\\tau(t)` by
69  radial basis functions as follows. Define
70 
71  .. math::
72 
73  x(t) = \\log t
74 
75  Depending whether ``function_type`` is set to ``1``, ``2``, or ``3``, one
76  of the following functions is defined:
77 
78  .. math::
79  :nowrap:
80 
81  \\begin{eqnarray}
82  y_1(t) &= \\frac{1}{\\tau(t)} - \\frac{1}{\\tau_0} - t, \\
83  y_2(t) &= \\frac{\\frac{1}{\\tau(t)}}{\\frac{1}{\\tau_0} + t} - 1, \\
84  y_3(t) &= 1 - \\tau(t) \\left( \\frac{1}{\\tau_0} + t \\right).
85  \\end{eqnarray}
86 
87  * The set of data :math:`(x, y_1(x))` are interpolated using
88  *cubic splines*.
89  * The set of data :math:`(x, y_2(x))` and :math:`(x, y_3(x))` are
90  interpolated using *Gaussian radial basis functions*.
91 
92  **Example**
93 
94  This class can be invoked from
95  :class:`imate.InterpolateTraceinv.InterpolateTraceinv` module using
96  ``method='RBF'`` argument.
97 
98  .. code-block:: python
99 
100  >>> from imate import generate_matrix
101  >>> from imate import InterpolateTraceinv
102 
103  >>> # Create a symmetric positive-definite matrix, size (20**2, 20**2)
104  >>> A = generate_matrix(size=20)
105 
106  >>> # Create an object that interpolates trace of inverse of A+tI
107  >>> # where I is identity matrix.
108  >>> TI = InterpolateTraceinv(A, method='RBF')
109 
110  >>> # Interpolate A+tI at some input point t
111  >>> t = 4e-1
112  >>> trace = TI.interpolate(t)
113  """

154  @staticmethod
155  def chebyshev_rational_roots(degree):
156  """
157  Returns Chebyshev points for Chebyshev rational function.
158  """
159 
160  # Chebyshev roots. x is an array of size degree.
161  x = scipy.special.roots_chebyt(degree)[0]
162 
163  # k = numpy.arange(degree)
164  # x = numpy.cos((2*k+1)*numpy.pi/(2*degree))
165 
166  # k = numpy.arange(1, degree)
167  # x = -numpy.cos(k*numpy.pi/n))
168 
169  x = numpy.sort(x)
170 
171  # Chebyshev rational roots. Convert x = (t-1)/(t+1) to t
172  t = (1.0+x) / (1.0-x)
173 
174  return t

180  def linear_system(self, scale, t_i, yi):
181  """
182  Creates a linear system T w = yi, where T_{ij} = T_[j](x_i) and
183  x_i = (t_i - scale) / (t_i + scale).
184 
185  * T is (q, q) matrix
186  * yi and ti are of size (q,)
187  * coefficients w is of the size (q,).
188  """
189 
190  scale = numpy.abs(scale)
191 
192  t_i_ = t_i / scale
193  xi = (t_i_ - 1.0) / (t_i_ + 1.0)
194 
195  if self.func_typefunc_type == 1:
196  xi = numpy.r_[-1.0, xi]
197  yi = numpy.r_[0.0, yi]
198 
199  T = numpy.zeros((xi.size, xi.size), dtype=float)
200  for i in range(xi.size):
201  for j in range(xi.size):
202  T[i, j] = 0.5 * (1.0 - eval_chebyt(
203  j+1, xi[i]))
204 
205  Tinv = numpy.linalg.inv(T)
206  coeff = numpy.dot(Tinv, yi)
207 
208  return coeff

214  def curvature_integral(self, scale, t_i, yi):
215  """
216  Computes the arc integral of the curvature squared.
217 
218  The function f is defined by
219 
220  f(x) = sum_{i=1}^q w_i * Ti(x)
221 
222  where the coefficients w_i are solved by fitting f(x_i) = yi, where
223  x_i = (t_i - scale) / (ti + scale)
224 
225  The arc curvature square integral is
226 
227  int_{-1}^{1} (f'')**2 / (1 + (f')**2)**(2.5) dx.
228  """

234  def d1T(n, x):
235  """
236  First derivative of Chebyshev polynomial of order n.
237  """
238 
239  dT_dx = n * eval_chebyu(n-1, x)
240  return dT_dx
241 
242  # ---
243  # d2T
244  # ---
245 
246  def d2T(n, x):
247  """
248  Second derivative of Chebyshev polynomial of order n.
249 
250  Second derivative of Chebyshev T_{n}(x) function is
251  n * (T_{n}(x) - x*U_{n-1}(x)) / (x**2-1) where U is the Chebyshev
252  function of the second kind.
253  """
254 
255  if x == 1.0:
256  d2T_dx2 = (n**4 - n**2) / 3.0
257  elif x == -1.0:
258  d2T_dx2 = (-1)**n * (n**4 - n**2) / 3.0
259  else:
260  d2T_dx2 = n * \
261  (n * eval_chebyt(n, x) - x * eval_chebyu(n-1, x)) / \
262  (x**2 - 1.0)
263 
264  return d2T_dx2

270  def arc_curvature_squared(x, coeff):
271  """
272  Returns the arc curvature squared of the function:
273 
274  f(x) = sum_{i=1}^q w_i * Ti(x)
275 
276  at point x. T_i is the Chebyshev polynomial of order i. The arc
277  curvature square is
278 
279  (f'')**2 / (1 + (f')**2)**(2.5)
280  """
281 
282  d1_poly = 0
283  for j in range(coeff.size):
284  d1_poly += 0.5 * coeff[j] * d1T(j+1, x)
285 
286  d2_poly = 0
287  for j in range(coeff.size):
288  d2_poly += 0.5 * coeff[j] * d2T(j+1, x)
289 
290  # Arc curvature squared
291  curv = d2_poly**2 / (1.0 + d1_poly**2)**(2.5)
292 
293  # Pure curvature squared
294  # curv = d2_poly**2
295 
296  return curv

356  Interpolates :math:`\\mathrm{trace} \\left( (\\mathbf{A} +

361  object is initialized.
