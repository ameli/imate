124 template <typename DataType>
125 void cOrthogonalization<DataType>::gram_schmidt_process(
126  const DataType* V,
127  const LongIndexType vector_size,
128  const IndexType num_vectors,
129  const IndexType last_vector,
130  const FlagType num_ortho,
131  DataType* v)
132 {
133  // Determine how many previous vectors to orthogonalize against
134  IndexType num_steps;
135  if ((num_ortho == 0) || (num_vectors < 2))
136  {
137  // No orthogonalization is performed
138  return;
139  }
140  else if ((num_ortho < 0) ||
141  (num_ortho > static_cast<FlagType>(num_vectors)))
142  {
143  // Orthogonalize against all vectors
144  num_steps = num_vectors;
145  }
146  else
147  {
148  // Orthogonalize against only the last num_ortho vectors
149  num_steps = num_ortho;
150  }
151 
152  // Vectors can be orthogonalized at most to the full basis of the vector
153  // space. Thus, num_steps cannot be larger than the dimension of vector
154  // space, which is vector_size.
155  if (num_steps > static_cast<IndexType>(vector_size))
156  {
157  num_steps = vector_size;
158  }
159 
160  IndexType i;
161  DataType inner_prod;
162  DataType norm;
163  DataType norm_v;
164  DataType epsilon = std::numeric_limits<DataType>::epsilon();
165  DataType distance;
166 
167  // Iterate over vectors
168  for (IndexType step=0; step < num_steps; ++step)
169  {
170  // i is the index of a column vector in V to orthogonalize v against it
171  if ((last_vector % num_vectors) >= step)
172  {
173  i = (last_vector % num_vectors) - step;
174  }
175  else
176  {
177  // Wrap around negative indices from the end of column index
178  i = (last_vector % num_vectors) - step + num_vectors;
179  }
180 
181  // Norm of j-th vector
182  norm = cVectorOperations<DataType>::euclidean_norm(
183  &V[vector_size*i], vector_size);
184 
185  // Check norm
186  if (norm < epsilon * sqrt(vector_size))
187  {
188  std::cerr << "WARNING: norm of the given vector is too small. " \
189  << "Cannot orthogonalize against zero vector. " \
190  << "Skipping." << std::endl;
191  continue;
192  }
193 
194  // Projection
195  inner_prod = cVectorOperations<DataType>::inner_product(
196  &V[vector_size*i], v, vector_size);
197 
198  // scale for subtraction
199  DataType scale = inner_prod / (norm * norm);
200 
201  // If scale is is 1, it is possible that vector v and j-th vector are
202  // identical (or close).
203  if (std::abs(scale - 1.0) <= 2.0 * epsilon)
204  {
205  // Norm of the vector v
206  norm_v = cVectorOperations<DataType>::euclidean_norm(
207  v, vector_size);
208 
209  // Compute distance between the j-th vector and vector v
210  distance = sqrt(norm_v*norm_v - 2.0*inner_prod + norm*norm);
211 
212  // If distance is zero, do not reorthogonalize i-th against
213  // the j-th vector.
214  if (distance < 2.0 * epsilon * sqrt(vector_size))
215  {
216  continue;
217  }
218  }
219 
220  // Subtraction
221  cVectorOperations<DataType>::subtract_scaled_vector(
222  &V[vector_size*i], vector_size, scale, v);
223  }
224 }

268 template <typename DataType>
269 void cOrthogonalization<DataType>::orthogonalize_vectors(
270  DataType* vectors,
271  const LongIndexType vector_size,
272  const IndexType num_vectors)
273 {
274  // Do nothing if there is only one vector
275  if (num_vectors < 2)
276  {
277  return;
278  }
279 
280  IndexType i = 0;
281  IndexType j;
282  IndexType start = 0;
283  DataType inner_prod;
284  DataType norm;
285  DataType norm_i;
286  DataType distance;
287  DataType epsilon = std::numeric_limits<DataType>::epsilon();
288  IndexType success = 1;
289  IndexType max_num_trials = 20;
290  IndexType num_trials = 0;
291  IndexType num_threads = 1;
292  RandomNumberGenerator random_number_generator(num_threads);
293 
294  while (i < num_vectors)
295  {
296  if ((success == 0) && (num_trials >= max_num_trials))
297  {
298  std::cerr << "ERROR: Cannot orthogonalize vectors after " \
299  << num_trials << " trials. Aborting." \
300  << std::endl;
301  abort();
302  }
303 
304  // Reset on new trial (if it was set to 0 before to start a new trial)
305  success = 1;
306 
307  // j iterates on previous vectors in a window of at most vector_size
308  if (static_cast<LongIndexType>(i) > vector_size)
309  {
310  // When vector_size is smaller than i, it is fine to cast to signed
311  start = i - static_cast<IndexType>(vector_size);
312  }
313 
314  // Reorthogonalize against previous vectors
315  for (j=start; j < i; ++j)
316  {
317  // Norm of the j-th vector
318  norm = cVectorOperations<DataType>::euclidean_norm(
319  &vectors[j*vector_size], vector_size);
320 
321  // Check norm
322  if (norm < epsilon * sqrt(vector_size))
323  {
324  std::cerr << "WARNING: norm of the given vector is too " \
325  << " small. Cannot reorthogonalize against zero" \
326  << "vector. Skipping."
327  << std::endl;
328  continue;
329  }
330 
331  // Projecting i-th vector to j-th vector
332  inner_prod = cVectorOperations<DataType>::inner_product(
333  &vectors[i*vector_size], &vectors[j*vector_size],
334  vector_size);
335 
336  // Scale of subtraction
337  DataType scale = inner_prod / (norm * norm);
338 
339  // If scale is is 1, it is possible that i-th and j-th vectors are
340  // identical (or close). So, instead of subtracting them,
341  // regenerate new i-th vector.
342  if (std::abs(scale - 1.0) <= 2.0 * epsilon)
343  {
344  // Norm of the i-th vector
345  norm_i = cVectorOperations<DataType>::euclidean_norm(
346  &vectors[i*vector_size], vector_size);
347 
348  // Compute distance between i-th and j-th vector
349  distance = sqrt(norm_i*norm_i - 2.0*inner_prod + norm*norm);
350 
351  // If distance is zero, do not reorthogonalize i-th against
352  // vector j-th and the subsequent vectors after j-th.
353  if (distance < 2.0 * epsilon * sqrt(vector_size))
354  {
355  // Regenerate new random vector for i-th vector
356  RandomArrayGenerator<DataType>::generate_random_array(
357  random_number_generator, &vectors[i*vector_size],
358  vector_size, num_threads);
359 
360  // Repeat the reorthogonalization for i-th vector against
361  // all previous vectors again.
362  success = 0;
363  ++num_trials;
364  break;
365  }
366  }
367 
368  // Subtraction
369  cVectorOperations<DataType>::subtract_scaled_vector(
370  &vectors[vector_size*j], vector_size, scale,
371  &vectors[vector_size*i]);
372 
373  // Norm of the i-th vector
374  norm_i = cVectorOperations<DataType>::euclidean_norm(
375  &vectors[i*vector_size], vector_size);
376 
377  // If the norm is too small, regenerate the i-th vector randomly
378  if (norm_i < epsilon * sqrt(vector_size))
379  {
380  // Regenerate new random vector for i-th vector
381  RandomArrayGenerator<DataType>::generate_random_array(
382  random_number_generator, &vectors[i*vector_size],
383  vector_size, num_threads);
384 
385  // Repeat the reorthogonalization for i-th vector against
386  // all previous vectors again.
387  success = 0;
388  ++num_trials;
389  break;
390  }
391  }
392 
393  if (success == 1)
394  {
395  ++i;
396 
397  // Reset if num_trials was incremented before.
398  num_trials = 0;
399  }
400  }
401 }
