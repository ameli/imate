24 class EigenvaluesMethod(InterpolantBase):
25  """
26  Note: B is always None (identity) in EIG method.
27 
28  Computes the trace of inverse of an invertible matrix :math:`\\mathbf{A} +
29  t \\mathbf{B}` using eigenvalues of :math:`\\mathbf{A}` and
30  :math:`\\mathbf{B}`.
31 
32  The trace of computed by
33 
34  .. math::
35 
36  \\mathrm{trace}\\left( (\\mathbf{A} + t \\mathbf{B})^{-1} \\right)
37  = \\sum_{i = 1}^n \\frac{1}{\\lambda_i + t \\mu_i}
38 
39  where :math:`\\lambda_i` is the eigenvalue of :math:`\\mathbf{A}`
40  and :math:`\\mu_i` is the eigenvalue of :math:`\\mathbf{B}`.
41  This class does not accept interpolant points as the result is not
42  interpolated.
43 
44  **Class Inheritance:**
45 
46  .. inheritance-diagram:: imate.InterpolateTraceinv.EigenvaluesMethod
47  :parts: 1
48 
49  :param A: Invertible matrix, can be either dense or sparse matrix.
50  :type A: numpy.ndarray
51 
52  :param B: Invertible matrix, can be either dense or sparse matrix.
53  :type B: numpy.ndarray
54 
55  :param options: A dictionary of input arguments for
56  :mod:`imate.traceinv.traceinv` module.
57  :type options: dict
58 
59  :param non_zero_ratio: The ratio of the number of eigenvalues to be assumed
60  non-zero over all eigenvalues. This option is only used for sparse
61  matrices where as assume some of eigenvalues are very small and we are
62  only interested in computing non-zero eigenvalues. In practice, it is
63  not possible to compute all eigenvalues of a large sparse matrix.
64  Default is ``0.9`` indicating to compute 90 percent of the eigenvalues
65  with the largest magnitude and assume the rest of the eigenvalues are
66  zero.
67  :type non_zero_ratio: int
68 
69  :param tol: tol of computing eigenvalues. This option is only
70  used for sparse matrices. Default value is ``1e-3``.
71  :type tol: float
72 
73  :param verbose: If ``True``, prints some information on the computation
74  process. Default is ``False``.
75  :type verbose: bool
76 
77  .. note::
78 
79  The input matrices :math:`\\mathbf{A}` and :math:`\\mathbf{B}` can be
80  either sparse or dense. In case of a **sparse matrix**, only some of
81  the eigenvalues with the largest magnitude is computed and the rest of
82  its eigenvalues is assumed to be negligible. The ratio of computed
83  eigenvalues over the total number of eigenvalues can be set by
84  ``non_zero_ratio``. The tolerance at which the eigenvalues are computed
85  can be set by ``tol``.
86 
87  **Example:**
88 
89  This class can be invoked from
90  :class:`imate.InterpolateTraceinv.InterpolateTraceinv` module using
91  ``method='EIG'`` argument.
92 
93  .. code-block:: python
94 
95  >>> from imate import generate_matrix
96  >>> from imate import InterpolateTraceinv
97 
98  >>> # Create a symmetric positive-definite matrix, size (20**2, 20**2)
99  >>> A = generate_matrix(size=20)
100 
101  >>> # Create an object that interpolates trace of inverse of A+tI
102  >>> # where I is identity matrix.
103  >>> TI = InterpolateTraceinv(A, InterpolatiionMethod='EIG')
104 
105  >>> # Interpolate A+tI at some input point t
106  >>> t = 4e-1
107  >>> trace = TI.interpolate(t)
108 
109  .. seealso::
110 
111  The result of the ``EIG`` method is identical with the exact method
112  ``EXT``, which is given by
113  :class:`imate.InterpolateTraceinv.ExactMethod`.
114  """

213  def interpolate(self, t):
214  """
215  Computes the function :math:`\\mathrm{trace}\\left( (\\mathbf{A} +
216  t \\mathbf{B})^{-1} \\right)` at the input point :math:`t` by
217 
218  .. math::
219 
220  \\mathrm{trace}\\left( (\\mathbf{A} + t \\mathbf{B})^{-1}
221  \\right) = \\sum_{i = 1}^n \\frac{1}{\\lambda_i + t \\mu_i}
222 
223  where :math:`\\lambda_i` is the eigenvalue of :math:`\\mathbf{A}`.
224  and :math:`\\mu_i` is the eigenvalue of :math:`\\mathbf{B}`.
225 
226  :param: t: An inquiry point, which can be a single number, or an array
227  of numbers.
228  :type t: float or numpy.array
229 
230  :return: The exact value of the trace of inverse of ``A + tB``.
231  :rtype: float
232  """
233 
234  # Compute trace using eigenvalues
235  eig_An = self.A_eigenvaluesA_eigenvalues + t * self.B_eigenvaluesB_eigenvalues
236 
237  if self.ppp == 0:
238  # Compute logdet
239  logdet_ = numpy.sum(numpy.log(numpy.abs(eig_An)))
240  schatten = numpy.exp(logdet_/self.nn)
241  else:
242  tracep = numpy.sum(eig_An**self.ppp)
243  schatten = (tracep/self.nn)**(1.0/self.ppp)
244 
245  return schatten
