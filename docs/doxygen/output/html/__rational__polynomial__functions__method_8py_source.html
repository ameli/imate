24 class RationalPolynomialFunctionsMethod(InterpolantBase):
25  """
26  Computes the trace of inverse of an invertible matrix :math:`\\mathbf{A} +
27  t \\mathbf{B}` using an interpolation scheme based on rational polynomial
28  functions (see details below).
29 
30  **Class Inheritance:**
31 
32  .. inheritance-diagram::
33  imate.InterpolateTraceinv.RationalPolynomialFunctionsMethod
34  :parts: 1
35 
36  :param A: Invertible matrix, can be either dense or sparse matrix.
37  :type A: numpy.ndarray
38 
39  :param B: Invertible matrix, can be either dense or sparse matrix.
40  :type B: numpy.ndarray
41 
42  :param options: A dictionary of input arguments for
43  :mod:`imate.traceinv.traceinv` module.
44  :type options: dict
45 
46  :param verbose: If ``True``, prints some information on the computation
47  process. Default is ``False``.
48  :type verbose: bool
49 
50  **Interpolation Method**
51 
52  Define the function
53 
54  .. math::
55 
56  \\tau(t) = \\frac{\\mathrm{trace}\\left( (\\mathbf{A} +
57  t \\mathbf{B})^{-1} \\right)}{\\mathrm{trace}(\\mathbf{B}^{-1})}
58 
59  and :math:`\\tau_0 = \\tau(0)`. Then, we approximate :math:`\\tau^{-1}(t)`
60  by
61 
62  .. math::
63 
64  \\tau(t) \\approx \\frac{t^q + a_{q-1} t^{q-1} + \\cdots + a_1 t +
65  a_0}{t^{q+1} + b_p t^q + \\cdots + b_1 t + b_0}
66 
67  where :math:`a_0 = b_0 \\tau_0`. The rest of coefficients are found by
68  solving a linear system using the function value at the interpolant points
69  :math:`\\tau_i = \\tau(t_i)`.
70 
71  .. note::
72 
73  The number of interpolant points :math:`q` in this module can only be
74  either :math:`q = 2` or :math:`q = 4`.
75 
76  **Example**
77 
78  This class can be invoked from
79  :class:`imate.InterpolateTraceinv.InterpolateTraceinv` module
80  using ``method='RPF'`` argument.
81 
82  .. code-block:: python
83 
84  >>> from imate import generate_matrix
85  >>> from imate import InterpolateTraceinv
86 
87  >>> # Create a symmetric positive-definite matrix, size (20**2, 20**2)
88  >>> A = generate_matrix(size=20)
89 
90  >>> # Create an object that interpolates trace of inverse of A+tI
91  >>> # where I is identity matrix.
92  >>> TI = InterpolateTraceinv(A, method='RPF')
93 
94  >>> # Interpolate A+tI at some input point t
95  >>> t = 4e-1
96  >>> trace = TI.interpolate(t)
97  """

124  def fit_rational_poly(self, t_i, tau_i, tau0):
125  """
126  Finds the coefficients :math:`(a_1, a_0, b_2, b_1, b_0)` of a Rational
127  polynomial of order 2 over 3,
128 
129  .. math::
130 
131  \\tau(t) \\approx \\frac{t^2 + a_{1} t + a_0}{t^3 + b_2 t^2 + b_1 t
132  + b_0}
133 
134  This is used when the number of interpolant points is :math:`q = 4`.
135 
136  :param t_i: Inquiry point.
137  :type t_i: float
138 
139  :param tau_i: The function value at the inquiry point :math:`t`
140  :type tau_i: float
141 
142  :param tau_0: The function value at :math:`t_0 = 0`.
143  :type tau_0: float
144  """
145 
146  # Length of t_i should be 2q+1
147  q = t_i.size // 2
148  if q != t_i.size/2:
149  print(t_i.size)
150  print(q)
151  raise ValueError('In rational polynomial interpolation method, ' +
152  'the number of interpolation points should be ' +
153  'even.')
154 
155  # Matrix of coefficients
156  C = numpy.zeros((2*q, 2*q), dtype=float)
157  c = numpy.zeros((2*q,), dtype=float)
158 
159  for i in range(2*q):
160  c[i] = tau_i[i] * t_i[i]**q - t_i[i]**(q+1)
161 
162  C[i, 0] = 1.0-tau_i[i]/tau0
163  for j in range(1, q+1):
164  C[i, j] = t_i[i]**j
165  for j in range(1, q):
166  C[i, q+j] = -tau_i[i] * t_i[i]**j
167 
168  # Condition number
169  if self.verboseverbose:
170  print('Condition number: %0.2e' % (numpy.linalg.cond(C)))
171 
172  # Solve with least square. NOTE: don't solve with numpy.linalg.solve.
173  x = numpy.linalg.solve(C, c)
174  x = list(x)
175  b = x[:q+1]
176  a = x[q+1:]
177  a0 = b[0]/tau0
178  a = [a0] + a
179 
180  # Output
181  numerator = [1] + b[::-1]
182  denominator = [1] + a[::-1]
183 
184  # Check poles
185  Poles = numpy.roots(denominator)
186  if numpy.any(Poles > 0):
187  print('denominator poles: ', end='')
188  print(Poles)
189  raise ValueError('rational_polynomial has positive poles.')
190 
191  return numerator, denominator
