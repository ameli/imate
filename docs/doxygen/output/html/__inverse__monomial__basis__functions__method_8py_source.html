24 class InverseMonomialBasisFunctionsMethod(InterpolantBase):
25  """
26  Computes the trace of inverse of an invertible matrix :math:`\\mathbf{A} +
27  t \\mathbf{B}` using an interpolation scheme based on inverse monomial
28  basis functions (see details below).
29 
30  **Class Inheritance:**
31 
32  .. inheritance-diagram::
33  imate.InterpolateTraceinv.InverseMonomialBasisFunctionsMethod
34  :parts: 1
35 
36  :param A: Invertible matrix, can be either dense or sparse matrix.
37  :type A: numpy.ndarray
38 
39  :param B: Invertible matrix, can be either dense or sparse matrix.
40  :type B: numpy.ndarray
41 
42  :param options: A dictionary of input arguments fo
43  :mod:`imate.traceinv.traceinv` module.
44  :type options: dict
45 
46  :param verbose: If ``True``, prints some information on the computation
47  process. Default is ``False``.
48  :type verbose: bool
49 
50  :param basis_functions_type: One of the types ``'NonOrthogonal'``,
51  ``'Orthogonal'`` and ``'Orthogonal2'``. Default is ``'orthogonal2'``.
52  :type basisFunctionsType: string
53 
54  **Interpolation Method**
55 
56  Define the function
57 
58  .. math::
59 
60  \\tau(t) = \\frac{\\mathrm{trace}\\left( (\\mathbf{A} +
61  t \\mathbf{B})^{-1} \\right)}{\\mathrm{trace}(\\mathbf{B}^{-1})}
62 
63  and :math:`\\tau_0 = \\tau(0)`. Then, we approximate :math:`\\tau^{-1}(t)`
64  by
65 
66  .. math::
67 
68  \\frac{1}{\\tau(t)} \\approx \\frac{1}{\\tau_0} + \\sum_{i = 0}^q w_i
69  \\phi_i(t),
70 
71  where :math:`\\phi_i` are some known basis functions, and :math:`w_i` are
72  the coefficients of the linear basis functions. The first coefficient is
73  set to :math:`w_{0} = 1` and the rest of the weights are to be found from
74  the known function values :math:`\\tau_i = \\tau(t_i)` at some given
75  interpolant points :math:`t_i`.
76 
77  **Basis Functions:**
78 
79  In this module, three kinds of basis functions which can be set by the
80  argument ``basis_functions_type``.
81 
82  When ``basis_functions_type`` is set to ``NonOrthogonal``, the basis
83  functions are the inverse of the monomial functions defined by
84 
85  .. math::
86 
87  \\phi_i(t) = t^{\\frac{1}{i+1}}, \\qquad i = 0, \\dots, q.
88 
89  When ``basis_functions_type`` is set to ``'Orthogonal'`` or
90  ``'Orthogonal2'``, the orthogonal form of the above basis functions are
91  used. Orthogonal basis functions are formed by the above non-orthogonal
92  functions as
93 
94  .. math::
95 
96  \\phi_i^{\\perp}(t) = \\alpha_i \\sum_{j=1}^i a_{ij} \\phi_j(t)
97 
98  The coefficients :math:`\\alpha_i` and :math:`a_{ij}` can be obtained by
99  the python package `Orthogoanl Functions
100  <https://ameli.github.io/Orthogonal-Functions>`_. These coefficients are
101  hard-coded in this function up to :math:`i = 9`. Thus, in this module, up
102  to nine interpolant points are supported.
103 
104  .. warning::
105 
106  The non-orthogonal basis functions can lead to ill-conditioned system
107  of equations for finding the weight coefficients :math:`w_i`. When the
108  number of interpolating points is large (such as :math:`q > 6`), it is
109  recommended to use the orthogonalized set of basis functions.
110 
111  .. note::
112 
113  The recommended basis function type is ``'Orthogonal2'``.
114 
115 
116  **Example**
117 
118  This class can be invoked from
119  :class:`imate.InterpolateTraceinv.InterpolateTraceinv` module using
120  ``method='RMBF'`` argument.
121 
122  .. code-block:: python
123 
124  >>> from imate import generate_matrix
125  >>> from imate import InterpolateTraceinv
126 
127  >>> # Create a symmetric positive-definite matrix, size (20**2, 20**2)
128  >>> A = generate_matrix(size=20)
129 
130  >>> # Create an object that interpolates trace of inverse of A+tI
131  >>> # where I is identity matrix.
132  >>> TI = InterpolateTraceinv(A, method='RMBF')
133 
134  >>> # Interpolate A+tI at some input point t
135  >>> t = 4e-1
136  >>> trace = TI.interpolate(t)
137 
138  .. seealso::
139 
140  The other class that provides interpolation with basis functions method
141  is :mod:`imate.InterpolateTraceinv.Monomialbasis_functionsMethod`.
142  """

452  def interpolate(self, t):
453  """
454  Interpolates :math:`\\mathrm{trace} \\left( (\\mathbf{A} +
455  t \\mathbf{B})^{-1} \\right)` at :math:`t`.
456 
457  This is the main interface function of this module and it is used after
458  the interpolation
459  object is initialized.
460 
461  :param t: The inquiry point(s).
462  :type t: float, list, or numpy.array
463 
464  :return: The interpolated value of the trace.
465  :rtype: float or numpy.array
466 
467  **Details:**
468 
469  Depending on the ``basis_functions_type``, the interpolation is as
470  follows:
471 
472  For ``'NonOrthogonal'`` basis:
473 
474  .. math::
475 
476  \\frac{1}{\\tau(t)} = \\frac{1}{\\tau_0} + t + \\sum_{j=1}^q w_j
477  \\phi_j(t).
478 
479  For ``'Orthogonal'`` and ``'Orthogonal2'`` bases:
480 
481  .. math::
482 
483  \\frac{1}{\\tau(t)} = \\frac{1}{\\tau_0} + t + \\sum_{j=1}^q w_j
484  \\phi_j(t).
485  """
486 
487  # Interpolation
488  if self.basis_functions_typebasis_functions_type == 'NonOrthogonal':
489 
490  S = 0.0
491  for j in range(self.qq):
492  S += self.ww[j] * self.basis_functionsbasis_functions(j, t)
493  tau = self.tau0tau0 + S + t
494 
495  elif self.basis_functions_typebasis_functions_type == 'Orthogonal':
496 
497  S = 0.0
498  for j in range(self.ww.size):
499  S += self.ww[j] * self.basis_functionsbasis_functions(j, t/self.scale_tscale_t)
500  tau = self.tau0tau0 + S
501 
502  elif self.basis_functions_typebasis_functions_type == 'Orthogonal2':
503 
504  S = 0.0
505  for j in range(self.qq):
506  S += self.ww[j] * self.basis_functionsbasis_functions(j, t/self.scale_tscale_t)
507  tau = self.tau0tau0 + S + t
508 
509  schatten = tau * self.schatten_Bschatten_B
510  return schatten
