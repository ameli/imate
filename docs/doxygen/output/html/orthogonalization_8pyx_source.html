23 # gram schmidt process
24 # ====================
25 
26 cdef void gram_schmidt_process(
27  const DataType* V,
28  const LongIndexType vector_size,
29  const IndexType num_vectors,
30  const FlagType ortho_depth,
31  DataType* v) nogil:
32  """
33  Modified Gram-Schmidt orthogonalization process to orthogonalize the vector
34  ``v`` against a subset of the column vectors in the array ``V``.
35 
36  ``V`` is 1D array of the length ``vector_size * num_vectors`` to represent
37  a 2D array of a set of ``num_vectors`` column vectors, each of the length
38  ``vector_size``. The length of ``v`` is also ``vector_size``.
39 
40  ``v`` is orthogonalized against the last ``ortho_depth`` columns of ``V``.
41 
42  * If ``ortho_depth`` is zero, no orthogonalization is performed.
43  * If ``ortho-depth`` is negative (usually set to ``-1``), then ``v`` is
44  orthogonalized against all column vectors of ``V``.
45  * If `ortho_depth`` is larger than ``num_vectors``, then ``v`` is
46  orthogonalized against all column vectors of ``V``.
47  * If ``ortho-depth`` is smaller than ``num_vectors``, then ``v`` is
48  orthogonalized against the last ``orth-depth`` column vectors of
49  ``V``.
50 
51  The result of the newer ``v`` is written in-place in ``v``.
52 
53  .. note::
54 
55  If two vectors are identical (or the norm of their difference is very
56  small), they cannot be orthogonalized against each other. In this case,
57  orthogonalization is skipped against the identical vector.
58 
59  :param V: 1D coalesced array of vectors representing a 2D array.
60  The length of this 1D array is vector_size*num_vectors, which indicates
61  a 2D array with the shape ``(vector_size, num_vectors)``.
62  :type V: C pointer (DataType)
63 
64  :param vector_size: The length of each vector. If we assume ``V`` indicates
65  a 2D vector, this the number of rows of ``V``.
66  :type vector_size: int
67 
68  :param num_vector: The number of column vectors. If we assume ``V``
69  indicates a 2D vector, this the number of columns of ``V``.
70  :type num_vectors: int
71 
72  :param ortho_depth: The number of vectors to be orthogonalized starting
73  from the last vector. ``0`` indicates no orthogonalization will be
74  performed and the function just returns. A negative value means all
75  vectors will be orthogonalized. A poisitive value will orthogonalize
76  the given number of vectors. This value cannot be larger than the
77  number of vectors.
78  :type ortho_depth: FlagType
79 
80  :param v: The vector that will be orthogonalized against the columns of
81  `V``. The length of ``v`` is ``vector_size``. This vector is modified
82  in-place.
83  :param v: C pointer (DataType)
84 
85  :param vector_size: The length of each vector. This the number of rows of
86  ``basis`` as a 2D array.
87  :type vector_size: int
88 
89  :param num_vector: The number of column vectors. This the number of columns
90  of ``basis`` as a 2D array.
91  :type num_vectors: int
92  """
93 
94  cdef IndexType i
95  cdef LongIndexType j
96  cdef IndexType num_steps
97  cdef DataType inner_prod
98  cdef DataType norm
99  cdef DataType norm_v
100  cdef DataType scale
101  cdef DataType epsilon = 1e-15
102 
103  # Determine how many previous vectors to orthogonalize against
104  if ortho_depth == 0:
105  # No orthogonalization is performed
106  return
107 
108  elif (ortho_depth < 0) or (ortho_depth > <FlagType> num_vectors):
109  # Orthogonalize against all vectors
110  num_steps = num_vectors
111 
112  else:
113  # Orthogonalize against only the last ortho_depth vectors
114  num_steps = ortho_depth
115 
116  # Vectors can be orthogonalized at most to the full basis of the vector
117  # space. Thus, num_steps cannot be larger than the dimension of vector
118  # space, which is vector_size.
119  if num_steps > <IndexType>(vector_size):
120  num_steps = vector_size
121 
122  # Iterate over vectors
123  for i in range(num_vectors-1, num_vectors-num_steps-1, -1):
124 
125  # Norm of j-th vector
126  norm = cVectorOperations[DataType].euclidean_norm(
127  &V[vector_size*i], vector_size)
128 
129  if norm < epsilon * sqrt(vector_size):
130  printf('WARNING: norm of the given vector is too small. Cannot ')
131  printf('reorthogonalize against zero vector. Skipping\n')
132  break
133 
134  # Projection
135  inner_prod = cVectorOperations[DataType].inner_product(
136  &V[vector_size*i], v, vector_size)
137 
138  # scale for subtraction
139  scale = inner_prod / (norm**2)
140 
141  # If scale is is 1, it is possible that vector v and j-th vector are
142  # identical (or close).
143  if fabs(scale - 1.0) <= 2.0 * epsilon:
144 
145  # Norm of the vector v
146  norm_v = cVectorOperations[DataType].euclidean_norm(v, vector_size)
147 
148  # Compute distance between the j-th vector and vector v
149  distance = sqrt(norm_v**2 - 2.0*inner_prod + norm**2)
150 
151  # If distance is zero, do not reorthogonalize i-th against
152  # the j-th vector.
153  if distance < 2.0 * epsilon * sqrt(vector_size):
154 
155  # Skip orthogonalizing against j-th vector
156  continue
157 
158  # Subtraction
159  cVectorOperations[DataType].subtract_scaled_vector(
160  &V[vector_size*i], vector_size, scale, v)

164 # orthogonalize vectors
165 # =====================
166 
167 cdef void orthogonalize_vectors(
168  DataType* vectors,
169  const LongIndexType vector_size,
170  const IndexType num_vectors) nogil:
171  """
172  Orthogonalizes set of vectors mutually using modified Gram-Schmidt process.
173 
174  .. note::
175 
176  Let ``m`` be the number of vectors (``num_vectors``), and let ``n`` be
177  the size of each vector (``vector_size``). In general, ``n`` is much
178  larger (large matrix size), and ``m`` is small, in order of a couple of
179  hundred. But for small matrices (where ``n`` could be smaller then
180  ``m``), then each vector can be orthogonalized at most to ``n`` other
181  vectors. This is because the dimension of the vector space is ``n``.
182  Thus, if there are extra vectors, each vector is orthogonalized to
183  window of the previous ``n`` vector.
184 
185  .. note::
186 
187  If two vectors are identical (or the norm of their difference is very
188  small), they cannot be orthogonalized against each other. In this case,
189  one of the vectors is re-generated by new random numbers.
190 
191  .. warning::
192 
193  if ``num_vectors`` is larger than ``vector_size``, the
194  orthogonalization fails since not all vectors are independent, and at
195  least one vector becomes zero.
196 
197  :param vectors: 2D array of size ``vector_size * num_vectors``. This array
198  will be modified in-place and will be output of this function. Note
199  this is Fortran ordering, meaning that the first index is contiguous.
200  Hence, to call the i-th vector, use ``&vectors[0][i]``. Here, iteration
201  over the first index is continuous.
202  :type vectors: pointer (DataType)
203 
204  :param vector_size: Number of rows of vectors array.
205  :type vector_size: IndexType
206 
207  :param num_vectors: Number of columns of vectors array.
208  :type num_vectors: unsigned long
209  """
210 
211  # Do nothing if there is only one vector
212  if num_vectors < 2:
213  return
214 
215  cdef IndexType i = 0
216  cdef IndexType j
217  cdef IndexType start_j
218  cdef DataType inner_prod
219  cdef DataType norm
220  cdef DataType norm_i
221  cdef DataType distance
222  cdef DataType scale
223  cdef DataType epsilon = 1e-15
224  cdef IndexType success = 1
225  cdef IndexType max_num_trials = 20
226  cdef IndexType num_trials = 0
227  cdef IndexType num_threads = 1
228 
229  while i < num_vectors:
230 
231  if success == 0 and num_trials >= max_num_trials:
232  printf('ERROR: Cannot orthogonalize vectors after %d trials.\n',
233  num_trials)
234  abort()
235 
236  # Reset on new trial (if it was set to 0 before to start a new trial)
237  success = 1
238 
239  # j iterates on previous vectors in a window of at most ``vector_size``
240  if <LongIndexType>i > vector_size:
241  start_j = i - vector_size
242  else:
243  start_j = 0
244 
245  # Reorthogonalize against previous vectors
246  for j in range(start_j, i):
247 
248  # Projecting i-th vector to j-th vector
249  inner_prod = cVectorOperations[DataType].inner_product(
250  &vectors[i*vector_size],
251  &vectors[j*vector_size],
252  vector_size)
253 
254  # Norm of the j-th vector
255  norm = cVectorOperations[DataType].euclidean_norm(
256  &vectors[j*vector_size], vector_size)
257 
258  # Check norm
259  if norm < epsilon * sqrt(vector_size):
260  printf('WARNING: norm of the given vector is too small. ')
261  printf('Cannot reorthogonalize against zero vector. ')
262  printf('Skipping.\n')
263  continue
264 
265  # Scale of subtraction
266  scale = inner_prod / (norm**2)
267 
268  # If scale is is 1, it is possible that i-th and j-th vectors are
269  # identical (or close). So, instead of subtracting them, regenerate
270  # a new i-th vector.
271  if fabs(scale - 1.0) <= 2.0 * epsilon:
272 
273  # Norm of the i-th vector
274  norm_i = cVectorOperations[DataType].euclidean_norm(
275  &vectors[i*vector_size], vector_size)
276 
277  # Compute distance between i-th and j-th vector
278  distance = sqrt(norm_i**2 - 2.0*inner_prod + norm**2)
279 
280  # If distance is zero, do not reorthogonalize i-th against
281  # vector j-th and the subsequent vectors after j-th.
282  if distance < 2.0 * epsilon * sqrt(vector_size):
283 
284  # Regenerate new random vector for i-th vector
285  with gil:
286  py_generate_random_array(&vectors[i*vector_size],
287  vector_size, num_threads)
288 
289  # Repeat the reorthogonalization for i-th vector against
290  # all previous vectors again.
291  success = 0
292  num_trials += 1
293  break
294 
295  # Subtraction
296  cVectorOperations[DataType].subtract_scaled_vector(
297  &vectors[vector_size*j], vector_size, scale,
298  &vectors[vector_size*i])
299 
300  # Check norm of the i-th vector
301  norm_i = cVectorOperations[DataType].euclidean_norm(
302  &vectors[i*vector_size], vector_size)
303 
304  # If the norm is too small, regenerate the i-th vector randomly
305  if norm_i < epsilon * sqrt(vector_size):
306 
307  # Regenerate new random vector for i-th vector
308  with gil:
309  py_generate_random_array(&vectors[i*vector_size],
310  vector_size, num_threads)
311 
312  # Repeat the reorthogonalization for i-th vector against
313  # all previous vectors again.
314  success = 0
315  num_trials += 1
316  break
317 
318  if success == 1:
319  i += 1
320 
321  # Reset if num_trials was incremented before.
322  num_trials = 0
